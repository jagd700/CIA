Lsizes = [120, 200, 300, 400, 600, 800, 1000, 1500, 2000]
df_NMOS = []

for i in range(0, len(Lsizes)):
        df_CSV = pd.read_csv('Resultados/NMOS_W5u_L' + str(Lsizes[i]) + 'n_Vds200m.csv')
        df_aux = pd.DataFrame()
        df_aux["gm_over_id"] = df_CSV["NM0:gm Y"]/df_CSV["NM0:id Y"]
        df_aux["gm_over_gds"] = df_CSV["NM0:gm Y"]/df_CSV["NM0:gds Y"]
        df_aux["gm_over_cgs_rad"] = df_CSV["NM0:gm Y"]/abs(df_CSV["NM0:cgs Y"])
        df_NMOS.append(df_aux)
           
plt.figure(1)
plt.title("NMOS gm/gds")
plt.ylabel("gm/gds")
plt.xlabel("gm/id")
for i in range(0, len(Lsizes)):
        plt.plot(df_NMOS[i]["gm_over_id"],df_NMOS[i]["gm_over_gds"], label="L=" + str(Lsizes[i]) + "nm")

plt.figure(2)
plt.title("NMOS gm/cgs rad")
plt.yscale('log')
plt.ylabel("gm/cgs rad")
plt.xlabel("gm/id")
for i in range(0, len(Lsizes)):
        plt.plot(df_NMOS[i]["gm_over_id"],df_NMOS[i]["gm_over_cgs_rad"], label="L=" + str(Lsizes[i]) + "nm")
